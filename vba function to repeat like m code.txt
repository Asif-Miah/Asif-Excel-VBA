Function LookupRemark(KeyValue As String, KeyRange As Range, RemarkRange As Range) As String
    Dim r As Long
    Dim totalRows As Long
    
    totalRows = KeyRange.Rows.Count
    
    'Return the first available remark for the matching key
    For r = 1 To totalRows
        If KeyRange.Cells(r, 1).Value = KeyValue Then
            If Trim(RemarkRange.Cells(r, 1).Value) <> "" Then
                LookupRemark = RemarkRange.Cells(r, 1).Value
                Exit Function
            End If
        End If
    Next r
    
    LookupRemark = ""  'No remark found
End Function
